    lib/models/DataTypes.js

    'use strict'

const STRING = require('../datatypes/STRING')
const INTEGER = require('../datatypes/INTEGER')

/**
* Describe una serie de funciones para crear diferentes tipos de datos
*/
class DataTypes {

  /**
  * Devuelve una instancia de la clase STRING.
  * @param {Number} [length=255] Longitud máxima de caracteres.
  * @return {STRING} Tipo de dato STRING.
  */
  static STRING(length = 255) {
    return new STRING(length)
  }

  /**
  * Devuelve una instancia de la clase INTEGER.
  * @return {INTEGER} Tipo de dato INTEGER.
  */
  static INTEGER() {
    return new INTEGER()
  }

}

module.exports = DataTypes
