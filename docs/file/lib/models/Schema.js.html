    'use strict'
const fs = require('fs')
const { GraphQLObjectType, GraphQLList, GraphQLString, GraphQLInt } = require('graphql')
const Util = require('../tools/Util')

class Schema {

  constructor(schemaPath, models, db) {
    this.path = {
      schema: schemaPath,
      query: {
        get: Util.absolutePath(schemaPath, './query/get'),
        getAll: Util.absolutePath(schemaPath, './query/getAll')
      },
      mutation: {
        create: Util.absolutePath(schemaPath, './mutation/create'),
        update: Util.absolutePath(schemaPath, './mutation/update'),
        delete: Util.absolutePath(schemaPath, './mutation/delete')
      },
      types: Util.absolutePath(schemaPath, './types'),
      inputs: Util.absolutePath(schemaPath, './inputs')
    }
    this.models = models
    this.db = db
    this.types = {}
    this.inputs = {}
  }

  addType(name) {
    let path = `${this.path.types}/${name}.js`
    if (this._existFile(path)) {
      let file = require(path)
      this.types[name] = file(this.models, this.db, this.types)
      return
    }
    if (!this.models[name]) {
      let msg = `No existe el modelo '${name}'`
      throw new Error(msg)
    }
    this.types[name] = this.models[name].graphQL(this)
  }

  addInput(name) {
    let path = `${this.path.inputs}/${name}.js`
    if (this._existFile(path)) {
      let file = require(path)
      this.inputs[name] = file(this.models, this.inputs)
      return
    }
    if (!this.models[name]) {
      let msg = `No existe el modelo '${name}'`
      throw new Error(msg)
    }
    this.inputs[name] = this.models[name].graphQLInput(name, this)
  }

  graphQL() {
    let schemaFile = require(Util.absolutePath(this.path.schema, './index.js'))
    return schemaFile(this)
  }

  queryGet(modelName) {
    let path = `${this.path.query.get}/${modelName}.js`
    if (this._existFile(path)) {
      let file = require(path)
      return file(this.models, this.db, this.types)
    }
    if (!this.models[modelName]) {
      let msg = `No existe el modelo '${modelName}'`
      throw new Error(msg)
    }
    return this.models[modelName].graphQLGet(this)
  }

  queryGetAll(modelName) {
    let path = `${this.path.query.getAll}/${modelName}.js`
    if (this._existFile(path)) {
      let file = require(path)
      return file(this.models, this.db, this.types)
    }
    if (!this.models[modelName]) {
      let msg = `No existe el modelo '${modelName}'`
      throw new Error(msg)
    }
    return this.models[modelName].graphQLGetAll(this)
  }

  mutationCreate(modelName) {
    let path = `${this.path.mutation.create}/${modelName}.js`
    if (this._existFile(path)) {
      let file = require(path)
      return file(this.models, this.db, this.types, this.inputs)
    }
    if (!this.models[modelName]) {
      let msg = `No existe el modelo '${modelName}'`
      throw new Error(msg)
    }
    return this.models[modelName].graphQLCreate(this)
  }

  mutationUpdate(modelName) {
    let path = `${this.path.mutation.update}/${modelName}.js`
    if (this._existFile(path)) {
      let file = require(path)
      return file(this.models, this.db, this.types, this.inputs)
    }
    if (!this.models[modelName]) {
      let msg = `No existe el modelo '${modelName}'`
      throw new Error(msg)
    }
    return this.models[modelName].graphQLUpdate(this)
  }

  mutationDelete(modelName) {
    let path = `${this.path.mutation.delete}/${modelName}.js`
    if (this._existFile(path)) {
      let file = require(path)
      return file(this.models, this.db, this.types)
    }
    if (!this.models[modelName]) {
      let msg = `No existe el modelo '${modelName}'`
      throw new Error(msg)
    }
    return this.models[modelName].graphQLDelete(this)
  }

  _existFile(filePath) {
    try {
      require(filePath)
      return true
    } catch(err) {
      if (err.message.startsWith('Cannot find module')) {
        return false
      }
      console.log(err)
      throw new Error(err)
    }
  }

}

module.exports = Schema
