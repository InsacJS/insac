    lib/models/DataType.js

    'use strict'

/**
* Describe el tipo de dato de un campo, sus caracteristicas y comportamiento.
*/
class DataType {

  /**
  * Crea una instancia de la clase DataType.
  * @param {String} name Nombre que identifica al tipo de dato.
  * @param {Array} args Argumentos necesarios de acuerdo al tipo de dato.
  */
  constructor(name, args) {
    /**
    * Nombre que identifica al tipo de dato.
    * @type {String}
    */
    this.name = name

    /**
    * Argumentos necesarios de acuerdo al tipo de dato.
    * @type {Array}
    */
    this.args = args
  }

  /**
  * Devuelve una instancia de la clase DataType cuyo tipo de dato es STRING.
  * @param {Number} [length=255] Longitud máxima de caracteres.
  * @return {DataType} Tipo de dato STRING.
  */
  static STRING(length = 255) {
    return new DataType()
  }

  /**
  * Devuelve una instancia de la clase DataType cuyo tipo de dato es INTEGER.
  * @return {DataType} Tipo de dato INTEGER.
  */
  static INTEGER() {
    return new DataType()
  }

}

module.exports = DataType
