    lib/tools/ModelManager.js

    'use strict'
/** @ignore */ const Reference = require('../models/Reference')

/**
* Contiene varias funciones para gestionar los modelos de la apliacción.
*/
class ModelManager {

  /**
  * Crea un objeto sequelize y actualiza las referencias de todos los modelos involucrados.
  * @param {!Model} model modelo a crear.
  * @param {!Database} database Instancia de la clase Database de la aplicación.
  * @param {!Model[]} models Modelos de la aplicación.
  */
  static addModel(model, database, models) {
    let modelName = model.name
    let define = model.sequelizeDefine()

    models[modelName] = model
    database.models[modelName] = database.sequelize.define(define.name, define.attributes, define.options)
    this._updateReferences(model, database, models)
  }

  /**
  * Actualiza las referencias de los modelos.
  * @param {!Model} model Modelo con referencias.
  * @param {!Database} database Instancia de la clase Database de la aplicación.
  * @param {!Model[]} models Modelos de la aplicación.
  */
  static _updateReferences(model, database, models) {
    let modelName = model.name
    for (let i in model.fields) {
      let field = model.fields[i]
      if (field instanceof Reference) {
        switch (field.reference.type) {
          case '1:1':
            models[field.reference.model].options.associations.push({model:model.name, as:model.options.singular})
            database.models[field.reference.model].hasOne(database.models[modelName], {as:model.options.singular, foreignKey: field.name})
            database.models[modelName].belongsTo(database.models[field.reference.model], {as:models[field.reference.model].options.singular, foreignKey: field.name})
            break
          case '1:N':
            models[field.reference.model].options.associations.push({model:model.name, as:model.options.plurl})
            database.models[field.reference.model].belongsTo(database.models[modelName], {as:model.options.plural, foreignKey: field.name})
            database.models[modelName].belongsTo(database.models[field.reference.model], {as:models[field.reference.model].options.singular, foreignKey: field.name})
            break
          default:
            let msg = `No se reconoce el tipo de referencia '${field.reference.type}' del modelo '${modelName}'`
            throw new Error(msg)
        }
      }
    }
  }

}

module.exports = ModelManager
